### 1.3 设计方案

#### A. 引入"时段锚点"概念

把"当前查看的时段"抽象为 `_anchorDate: MutableStateFlow<LocalDate>`，初始值 `LocalDate.now()`。`DateRange.statsRange(range, anchor)` 计算 `[from, to]`（把现有 util 的入参 `today` 改名 `anchor`，语义不变）。

```kotlin
class StatsViewModel(application: Application) : AndroidViewModel(application) {
    private val _range = MutableStateFlow(StatsRange.WEEK)
    private val _anchorDate = MutableStateFlow(LocalDate.now())   // 新增

    val uiState = combine(_range, _anchorDate) { range, anchor -> range to anchor }
        .flatMapLatest { (range, anchor) ->
            val (from, to) = DateRange.statsRange(range, anchor)
            combine(entryDao.observeByDateRange(from, to), categoriesFlow) { entries, cats ->
                computeStats(entries, cats)
            }
        }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), StatsUiState())

    fun shiftPeriod(delta: Int) { /* 见 B */ }
    fun setRange(r: StatsRange) { /* 见 C */ }
}
```

#### B. Dense Pager（Q2 已敲定）

提前算出所有"有时段数据"的 anchor 列表，`pageCount = validAnchors.size`。空时段根本不存在页，"直接跳过"天然成立。

**新 DAO**：

```kotlin
@Query("SELECT DISTINCT date FROM time_entries ORDER BY date DESC")
fun observeAllDates(): Flow<List<LocalDate>>
```

**ViewModel**：

```kotlin
@OptIn(ExperimentalCoroutinesApi::class)
val validAnchors: StateFlow<List<LocalDate>> = _range
    .flatMapLatest { range ->
        entryDao.observeAllDates().map { dates ->
            dates.map { anchorForRange(range, it) }
                .distinct()
                .sortedDescending()      // 最新在前，idx=0 = 最近有数据时段
        }
    }
    .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

private fun anchorForRange(range: StatsRange, date: LocalDate): LocalDate = when (range) {
    StatsRange.TODAY -> date
    StatsRange.WEEK  -> date.with(DayOfWeek.MONDAY)
    StatsRange.MONTH -> date.withDayOfMonth(1)
    StatsRange.YEAR  -> LocalDate.of(date.year, 1, 1)
}
```

**当前页管理（按 anchor 记忆——Q6 倾向 B）**：

```kotlin
private val _selectedIdx = MutableStateFlow(0)

val anchorDate: StateFlow<LocalDate> = combine(validAnchors, _selectedIdx) { anchors, idx ->
    anchors.getOrNull(idx) ?: LocalDate.now()
}.stateIn(...)

fun shift(delta: Int) {
    val size = validAnchors.value.size
    if (size == 0) return
    // 按 anchor 记忆：若新增 entry 导致列表前面长了项，仍指向同一 anchor
    val currentAnchor = validAnchors.value.getOrNull(_selectedIdx.value)
    _selectedIdx.value = (_selectedIdx.value + delta).coerceIn(0, size - 1)
    // 实施细节：entry 增删导致 anchors 列表变化时，重算 idx = anchors.indexOf(currentAnchor)
}

fun setRange(r: StatsRange) {
    if (_range.value == r) return
    _range.value = r
    _selectedIdx.value = 0   // 切 range 后回到最新
}
```

**实施备忘**：`observeAllDates` 的每次 emit 都会触发 `validAnchors` 重组。若新增 entry 让"本周"突然出现，列表前面会长 1 项；此时若 `_selectedIdx` 不变，用户会被动跳页。**修正**：维护 `_currentAnchor: MutableStateFlow<LocalDate?>`，每次 `validAnchors` 更新后 `idx = anchors.indexOf(_currentAnchor)` 取 `coerceIn(0, size-1)`；找不到则归零。

#### C. range 切换行为（Q3 倾向归零）

切 range → `_selectedIdx = 0` → 锚定最新有数据时段。简单且符合"切到 MONTH 看本月"的直觉。若"本月"无数据，自动落到最近有数据的月（idx=0 即列表第一项）。

#### D. UI 改造

```kotlin
@Composable
fun StatsScreen(viewModel: StatsViewModel) {
    val uiState by viewModel.uiState.collectAsState()
    val range by viewModel.range.collectAsState()
    val anchors by viewModel.validAnchors.collectAsState()
    val anchorDate by viewModel.anchorDate.collectAsState()
    val currentReview by viewModel.currentReview.collectAsState()

    if (anchors.isEmpty()) {
        StatsEmptyState()   // 整库无任何 entry
        return
    }

    val pagerState = rememberPagerState(initialPage = 0, pageCount = { anchors.size })

    LaunchedEffect(pagerState) {
        snapshotFlow { pagerState.currentPage }
            .distinctUntilChanged()
            .collect { page -> viewModel.selectPage(page) }
    }
    LaunchedEffect(viewModel) {
        snapshotFlow { viewModel.selectedIdx.value }
            .distinctUntilChanged()
            .collect { idx ->
                if (pagerState.currentPage != idx) pagerState.animateScrollToPage(idx)
            }
    }

    Column(modifier = Modifier.fillMaxSize()) {
        StatsHeader(totalMin = uiState.totalMin, periodText = formatPeriod(range, anchorDate))
        HorizontalDivider()
        FlowRow { /* range chips, onClick = viewModel.setRange */ }
        HorizontalDivider()
        HorizontalPager(state = pagerState, modifier = Modifier.fillMaxSize()) { page ->
            StatsPageContent(
                statsGroups = uiState.groups,
                review = currentReview,
                range = range,
                periodLabel = periodLabelFor(range)
            )
        }
    }
}
```

每个 page 渲染的是 VM 当前 uiState（由 `anchorDate` 推出）。Pager 切换 → `selectPage` 改 idx → `anchorDate` 变 → uiState + currentReview 重组。

#### E. 跨年 / 跨月边界

`anchor = monday / day1 / jan1` 统一规则，跨年周（如 2025-12-29 周一 ~ 2026-01-04 周日）属于 anchor=2025-12-29 那一周，在 YEARS 视角下也归 2025 年。无歧义，无需特殊处理。

#### F. Date picker 集成（Q4 倾向不加）

滑动已够用；TODAY 屏已有 picker，STATS 不必重复。

```kotlin
            put("recentReviews", JSONArray().apply {
                recentReviews.forEach { r -> put(JSONObject().apply {
                    put("periodStart", r.periodStart.toString())
                    put("periodEnd", r.periodEnd.toString())
                    put("summary", r.summaryText.orEmpty())
                    put("findings", r.mainFindings.orEmpty())
                    put("adjust", r.adjustmentPlan.orEmpty())
                }) })
            })
        }
    }
    return JSONArray().apply {
        put(JSONObject().put("role", "system").put("content", effectivePrompt + "\n\n" + JSON_FORMAT_SUFFIX))
        put(JSONObject().put("role", "user").put("content", userObj.toString()))
    }
}
```

**流程**（伪代码）：

```kotlin
fun generateWithAi() = viewModelScope.launch {
    if (_aiState.value is AiState.Loading) return
    val period = _periodType.value

private suspend fun proceedAiCall(period, today, selectedMondays) {
    _aiState.value = AiState.Loading
    val key = settingDao.getByKey(KEY_DEEPSEEK_API_KEY)?.trim()
    if (key.isNullOrBlank()) { _aiState.value = AiState.Error("..."); return }
    val model = settingDao.getByKey(KEY_DEEPSEEK_MODEL) ?: DEFAULT_DEEPSEEK_MODEL
    val userPrompt = settingDao.getByKey(KEY_DEEPSEEK_PROMPT) ?: DEFAULT_DEEPSEEK_PROMPT
    val cfg = DeepSeekConfig(apiKey = key, model = model)

    val currentRange = DateRange.currentFor(period, today)
    val entries = entryDao.getByDateRange(currentRange.first, currentRange.second)
    if (entries.isEmpty()) {
        _aiState.value = AiState.Error("本${period.label}暂无时间记录，无法生成")
        return
    }
    val categories = categoryDao.getAll()
    val dayStart = settingDao.getByKey(KEY_DAY_START_MIN)?.toIntOrNull() ?: 480

    val periodWord = when (period) { DAY -> "日"; WEEK -> "周"; MONTH -> "月" }
    val effectivePrompt = userPrompt.replace("{period}", periodWord)
    val recentReviews = fetchRecentReviews(period, today, selectedMondays)
    val messages = AiReviewRequestBuilder.buildMessages(
        period, currentRange, entries, categories, recentReviews, dayStart, effectivePrompt
    )
    when (val r = DeepSeekApi(cfg).chat(messages, jsonMode = true, maxTokens = 2000)) {
        is DeepSeekApi.Result.Success -> {
            val parsed = AiReviewRequestBuilder.parseResponse(r.content)
            if (parsed == null) _aiState.value = AiState.Error("AI 响应解析失败")
            else {
                _form.value = _form.value.copy(
                    summaryText = parsed.summary,
                    mainFindings = parsed.findings,
                    adjustmentPlan = parsed.adjust
                )
                _aiState.value = AiState.Done
            }
        }
        is DeepSeekApi.Result.Error ->
            _aiState.value = AiState.Error("API key 无效或网络错误：${r.message}")
    }
}

private suspend fun fetchRecentReviews(
    period: ReviewPeriod,
    today: LocalDate,
    selectedMondays: List<LocalDate>?
): List<ReviewEntity> = when (period) {
    ReviewPeriod.DAY -> {
        val n = settingDao.getByKey(KEY_AI_CONTEXT_DAY_DAYS)?.toIntOrNull() ?: 3
        (0 until n).mapNotNull { offset ->
            val d = today.minusDays(offset.toLong())
            reviewDao.getByPeriod(ReviewPeriod.DAY.name, d, d)
        }
    }
    ReviewPeriod.WEEK -> {
        val n = settingDao.getByKey(KEY_AI_CONTEXT_WEEK_DAYS)?.toIntOrNull() ?: 7
        val monday = today.with(DayOfWeek.MONDAY)
        val daysFromMonday = ChronoUnit.DAYS.between(monday, today).toInt() + 1
        val effectiveN = minOf(n, daysFromMonday)
        (0 until effectiveN).mapNotNull { offset ->
            val d = today.minusDays(offset.toLong())
            reviewDao.getByPeriod(ReviewPeriod.DAY.name, d, d)
        }
    }
    ReviewPeriod.MONTH -> {
        val mondays = selectedMondays ?: buildMonthWeekOptions(today)
            .filter { it.review != null }
            .map { it.monday }
        mondays.mapNotNull { monday ->
            reviewDao.getByPeriod(ReviewPeriod.WEEK.name, monday, monday.plusDays(6))
        }
    }
}

private suspend fun buildMonthWeekOptions(today: LocalDate): List<MonthWeekOption> {
    val monthStart = today.withDayOfMonth(1)
    val monthEnd = today.withDayOfMonth(today.lengthOfMonth())
    val mondays = (0..today.lengthOfMonth()).mapNotNull { offset ->
        val d = monthStart.plusDays(offset.toLong())
        if (d > monthEnd) null else d.with(DayOfWeek.MONDAY)
    }.distinct()
    return mondays.map { monday ->
        val review = reviewDao.getByPeriod(ReviewPeriod.WEEK.name, monday, monday.plusDays(6))
        MonthWeekOption(monday, monday.plusDays(6), review, selected = review != null)
    }
}
```
